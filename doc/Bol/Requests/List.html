def proxy
  @proxy ||= begin
    if @query.categories?
      Proxy.new(self, Parsers::Categories.new(self))
    elsif @query.refinements?
      Proxy.new(self, Parsers::Refinements.new(self))
    elsif @query.products?
      super
    end
  end
end
